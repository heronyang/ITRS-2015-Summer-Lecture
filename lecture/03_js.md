## IF / LOOP

    if (a == 'hello') {
        console.log("hi");
    } else {
        console.log("bye");
    }

    var n = 10;
    while(n >= 1) {
        console.log(n);
        n--;
    }

    for(var i=1 ; i<=10 ; i++) {
        console.log(i);
    }
